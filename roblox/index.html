Basic Script Structure


-- This is a comment
local function myFunction()
    print("Hello, Roblox!")
end

myFunction()

Variables


local myNumber = 42
local myString = "Hello, Roblox!"
local myBoolean = true

Conditional Statements


if myNumber > 50 then
    print("The number is greater than 50")
elseif myNumber < 50 then
    print("The number is less than 50")
else
    print("The number is equal to 50")
end

Loops


-- For loop
for i = 1, 5 do
    print(i)
end

-- While loop
local count = 0
while count < 5 do
    print(count)
    count = count + 1
end

Functions


local function greet(name)
    print("Hello, " .. name .. "!")
end

greet("Roblox")

Tables


local myTable = {
    "apple",
    "banana",
    "cherry"
}

print(myTable[1]) -- Outputs: apple

local myDictionary = {
    name = "John",
    age = 30
}

print(myDictionary.name) -- Outputs: John
